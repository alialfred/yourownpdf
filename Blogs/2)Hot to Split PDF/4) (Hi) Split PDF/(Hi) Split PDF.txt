async function executeClientSideSplit(arrayBuffer, splitPoints) {
  const sourceDoc = await PDFLib.PDFDocument.load(arrayBuffer);
  const segments = [];
  
  for (const range of splitPoints) {
    const newDoc = await PDFLib.PDFDocument.create();
    const copiedPages = await newDoc.copyPages(sourceDoc, range);
    copiedPages.forEach(page => newDoc.addPage(page));
    const pdfBytes = await newDoc.save();
    segments.push(pdfBytes);
  }
  return segments;
}
